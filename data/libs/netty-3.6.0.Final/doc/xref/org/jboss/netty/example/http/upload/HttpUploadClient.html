60      public void run() {
61          String postSimple, postFile, get;
62          if (baseUri.endsWith("/")) {
63              postSimple = baseUri + "formpost";
64              postFile = baseUri + "formpostmultipart";
65              get = baseUri + "formget";
66          } else {
67              postSimple = baseUri + "/formpost";
68              postFile = baseUri + "/formpostmultipart";
69              get = baseUri + "/formget";
70          }
71          URI uriSimple;
72          try {
73              uriSimple = new URI(postSimple);
74          } catch (URISyntaxException e) {
75              logger.error("Invalid URI syntax" + e.getCause());
76              return;
77          }
78          String scheme = uriSimple.getScheme() == null? "http" : uriSimple.getScheme();
79          String host = uriSimple.getHost() == null? "localhost" : uriSimple.getHost();
80          int port = uriSimple.getPort();
81          if (port == -1) {
82              if ("http".equalsIgnoreCase(scheme)) {
83                  port = 80;
84              } else if ("https".equalsIgnoreCase(scheme)) {
85                  port = 443;
86              }
87          }
88  
89          if (!"http".equalsIgnoreCase(scheme) && !"https".equalsIgnoreCase(scheme)) {
90              logger.error("Only HTTP(S) is supported.");
91              return;
92          }
93  
94          boolean ssl = "https".equalsIgnoreCase(scheme);
95  
96          URI uriFile;
97          try {
98              uriFile = new URI(postFile);
99          } catch (URISyntaxException e) {
100             logger.error("Error: " + e.getMessage());
101             return;
102         }
103         File file = new File(filePath);
104         if (! file.canRead()) {
105             logger.error("A correct path is needed");
106             return;
107         }
108 
109         // Configure the client.
110         ClientBootstrap bootstrap = new ClientBootstrap(
111                 new NioClientSocketChannelFactory(
112                         Executors.newCachedThreadPool(),
113                         Executors.newCachedThreadPool()));
114 
115         // Set up the event pipeline factory.
116         bootstrap.setPipelineFactory(new HttpUploadClientPipelineFactory(ssl));
117 
118         // setup the factory: here using a mixed memory/disk based on size threshold
119         HttpDataFactory factory = new DefaultHttpDataFactory(
120                 DefaultHttpDataFactory.MINSIZE); // Disk if size exceed MINSIZE
121         DiskFileUpload.deleteOnExitTemporaryFile = true; // should delete file on exit (in normal exit)
122         DiskFileUpload.baseDirectory = null; // system temp directory
123         DiskAttribute.deleteOnExitTemporaryFile = true; // should delete file on exit (in normal exit)
124         DiskAttribute.baseDirectory = null; // system temp directory
125 
126         // Simple Get form: no factory used (not usable)
127         List<Entry<String, String>> headers =
128             formget(bootstrap, host, port, get, uriSimple);
129         if (headers == null) {
130             factory.cleanAllHttpDatas();
131             return;
132         }
133         // Simple Post form: factory used for big attributes
134         List<InterfaceHttpData> bodylist =
135             formpost(bootstrap, host, port, uriSimple, file, factory, headers);
136         if (bodylist == null) {
137             factory.cleanAllHttpDatas();
138             return;
139         }
140         // Multipart Post form: factory used
141         formpostmultipart(bootstrap, host, port, uriFile, factory, headers, bodylist);
142 
143         // Shut down executor threads to exit.
144         bootstrap.releaseExternalResources();
145         // Really clean all temporary files if they still exist
146         factory.cleanAllHttpDatas();
147     }

149     /**
150      * Standard usage of HTTP API in Netty without file Upload (get is not able to achieve File upload
151      * due to limitation on request size).
152      * @return the list of headers that will be used in every example after
153     **/
154     private static List<Entry<String, String>> formget(ClientBootstrap bootstrap, String host, int port, String get,
155             URI uriSimple) {
156         // XXX /formget
157         // No use of HttpPostRequestEncoder since not a POST
158         // Start the connection attempt.
159         ChannelFuture future = bootstrap.connect(new InetSocketAddress(host, port));
160         // Wait until the connection attempt succeeds or fails.
161         Channel channel = future.awaitUninterruptibly().getChannel();
162         if (!future.isSuccess()) {
163             future.getCause().printStackTrace();
164             bootstrap.releaseExternalResources();
165             return null;
166         }
167 
168         // Prepare the HTTP request.
169         QueryStringEncoder encoder = new QueryStringEncoder(get);
170         // add Form attribute
171         encoder.addParam("getform", "GET");
172         encoder.addParam("info", "first value");
173         encoder.addParam("secondinfo", "secondvalue ���&");
174         // not the big one since it is not compatible with GET size
175         // encoder.addParam("thirdinfo", textArea);
176         encoder.addParam("thirdinfo", "third value\r\ntest second line\r\n\r\nnew line\r\n");
177         encoder.addParam("Send", "Send");
178 
179         URI uriGet;
180         try {
181             uriGet = new URI(encoder.toString());
182         } catch (URISyntaxException e) {
183             logger.error("Error: " + e.getMessage());
184             bootstrap.releaseExternalResources();
185             return null;
186         }
187 
188         HttpRequest request = new DefaultHttpRequest(
189                 HttpVersion.HTTP_1_1, HttpMethod.GET, uriGet.toASCIIString());
190         request.setHeader(HttpHeaders.Names.HOST, host);
191         request.setHeader(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.CLOSE);
192         request.setHeader(HttpHeaders.Names.ACCEPT_ENCODING, HttpHeaders.Values.GZIP + ',' +
193                 HttpHeaders.Values.DEFLATE);
194 
195         request.setHeader(HttpHeaders.Names.ACCEPT_CHARSET, "ISO-8859-1,utf-8;q=0.7,*;q=0.7");
196         request.setHeader(HttpHeaders.Names.ACCEPT_LANGUAGE, "fr");
197         request.setHeader(HttpHeaders.Names.REFERER, uriSimple.toString());
198         request.setHeader(HttpHeaders.Names.USER_AGENT, "Netty Simple Http Client side");
199         request.setHeader(HttpHeaders.Names.ACCEPT,
200                 "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8");
201         //connection will not close but needed
202         // request.setHeader("Connection","keep-alive");
203         // request.setHeader("Keep-Alive","300");
204 
205         CookieEncoder httpCookieEncoder = new CookieEncoder(false);
206         httpCookieEncoder.addCookie("my-cookie", "foo");
207         httpCookieEncoder.addCookie("another-cookie", "bar");
208         request.setHeader(HttpHeaders.Names.COOKIE, httpCookieEncoder.encode());
209 
210         List<Entry<String, String>> headers = request.getHeaders();
211         // send request
212         channel.write(request);
213 
214         // Wait for the server to close the connection.
215         channel.getCloseFuture().awaitUninterruptibly();
216 
217         return headers;
218     }

220     /**
221      * Standard post without multipart but already support on Factory (memory management)
222      *
223      * @return the list of HttpData object (attribute and file) to be reused on next post
224      */
225     private static List<InterfaceHttpData> formpost(ClientBootstrap bootstrap,
226             String host, int port,
227             URI uriSimple, File file, HttpDataFactory factory,
228             List<Entry<String, String>> headers) {
229         // XXX /formpost
230         // Start the connection attempt.
231         ChannelFuture future = bootstrap.connect(new InetSocketAddress(host, port));
232         // Wait until the connection attempt succeeds or fails.
233         Channel channel = future.awaitUninterruptibly().getChannel();
234         if (!future.isSuccess()) {
235             future.getCause().printStackTrace();
236             bootstrap.releaseExternalResources();
237             return null;
238         }
239 
240         // Prepare the HTTP request.
241         HttpRequest request = new DefaultHttpRequest(
242                 HttpVersion.HTTP_1_1, HttpMethod.POST, uriSimple.toASCIIString());
243 
244         // Use the PostBody encoder
245         HttpPostRequestEncoder bodyRequestEncoder = null;
246         try {
247             bodyRequestEncoder = new HttpPostRequestEncoder(factory,
248                     request, false);  // false => not multipart
249         } catch (NullPointerException e) {
250             // should not be since args are not null
251             e.printStackTrace();
252         } catch (ErrorDataEncoderException e) {
253             // test if method is a POST method
254             e.printStackTrace();
255         }
256 
257         // it is legal to add directly header or cookie into the request until finalize
258         for (Entry<String, String> entry : headers) {
259             request.setHeader(entry.getKey(), entry.getValue());
260         }
261 
262         // add Form attribute
263         try {
264             bodyRequestEncoder.addBodyAttribute("getform", "POST");
265             bodyRequestEncoder.addBodyAttribute("info", "first value");
266             bodyRequestEncoder.addBodyAttribute("secondinfo", "secondvalue ���&");
267             bodyRequestEncoder.addBodyAttribute("thirdinfo", textArea);
268             bodyRequestEncoder.addBodyFileUpload("myfile", file, "application/x-zip-compressed", false);
269             bodyRequestEncoder.addBodyAttribute("Send", "Send");
270         } catch (NullPointerException e) {
271             // should not be since not null args
272             e.printStackTrace();
273         } catch (ErrorDataEncoderException e) {
274             // if an encoding error occurs
275             e.printStackTrace();
276         }
277 
278         // finalize request
279         try {
280             request = bodyRequestEncoder.finalizeRequest();
281         } catch (ErrorDataEncoderException e) {
282             // if an encoding error occurs
283             e.printStackTrace();
284         }
285         // Create the bodylist to be reused on the last version with Multipart support
286         List<InterfaceHttpData> bodylist = bodyRequestEncoder.getBodyListAttributes();
287 
288         // send request
289         channel.write(request);
290 
291         // test if request was chunked and if so, finish the write
292         if (bodyRequestEncoder.isChunked()) { // could do either request.isChunked()
293             // either do it through ChunkedWriteHandler
294             channel.write(bodyRequestEncoder).awaitUninterruptibly();
295         }
296 
297         // Do not clear here since we will reuse the InterfaceHttpData on the next request
298         // for the example (limit action on client side). Take this as a broadcast of the same
299         // request on both Post actions.
300         //
301         // On standard program, it is clearly recommended to clean all files after each request
302         // bodyRequestEncoder.cleanFiles();
303 
304         // Wait for the server to close the connection.
305         channel.getCloseFuture().awaitUninterruptibly();
306         return bodylist;
307     }

309     /**
310      * Multipart example
311      */
312     private static void formpostmultipart(ClientBootstrap bootstrap, String host, int port,
313             URI uriFile, HttpDataFactory factory,
314             List<Entry<String, String>> headers, List<InterfaceHttpData> bodylist) {
315         // XXX /formpostmultipart
316         // Start the connection attempt.
317         ChannelFuture future = bootstrap.connect(new InetSocketAddress(host, port));
318         // Wait until the connection attempt succeeds or fails.
319         Channel channel = future.awaitUninterruptibly().getChannel();
320         if (!future.isSuccess()) {
321             future.getCause().printStackTrace();
322             bootstrap.releaseExternalResources();
323             return;
324         }
325 
326         // Prepare the HTTP request.
327         HttpRequest request = new DefaultHttpRequest(
328                 HttpVersion.HTTP_1_1, HttpMethod.POST, uriFile.toASCIIString());
329 
330         // Use the PostBody encoder
331         HttpPostRequestEncoder bodyRequestEncoder = null;
332         try {
333             bodyRequestEncoder = new HttpPostRequestEncoder(factory,
334                     request, true); // true => multipart
335         } catch (NullPointerException e) {
336             // should not be since no null args
337             e.printStackTrace();
338         } catch (ErrorDataEncoderException e) {
339             // test if method is a POST method
340             e.printStackTrace();
341         }
342 
343         // it is legal to add directly header or cookie into the request until finalize
344         for (Entry<String, String> entry : headers) {
345             request.setHeader(entry.getKey(), entry.getValue());
346         }
347 
348         // add Form attribute from previous request in formpost()
349         try {
350             bodyRequestEncoder.setBodyHttpDatas(bodylist);
351         } catch (NullPointerException e1) {
352             // should not be since previously created
353             e1.printStackTrace();
354         } catch (ErrorDataEncoderException e1) {
355             // again should not be since previously encoded (except if an error occurs previously)
356             e1.printStackTrace();
357         }
358 
359         // finalize request
360         try {
361             bodyRequestEncoder.finalizeRequest();
362         } catch (ErrorDataEncoderException e) {
363             // if an encoding error occurs
364             e.printStackTrace();
365         }
366 
367         // send request
368         channel.write(request);
369 
370         // test if request was chunked and if so, finish the write
371         if (bodyRequestEncoder.isChunked()) {
372             channel.write(bodyRequestEncoder).awaitUninterruptibly();
373         }
374 
375         // Now no more use of file representation (and list of HttpData)
376         bodyRequestEncoder.cleanFiles();
377 
378         // Wait for the server to close the connection.
379         channel.getCloseFuture().awaitUninterruptibly();
380     }
