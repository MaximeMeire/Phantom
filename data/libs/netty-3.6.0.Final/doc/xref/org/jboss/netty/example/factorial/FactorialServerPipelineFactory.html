16  package org.jboss.netty.example.factorial;
17  
18  import static org.jboss.netty.channel.Channels.*;
19  
20  import org.jboss.netty.channel.ChannelPipeline;
21  import org.jboss.netty.channel.ChannelPipelineFactory;
22  import org.jboss.netty.handler.codec.compression.ZlibDecoder;
23  import org.jboss.netty.handler.codec.compression.ZlibEncoder;
24  import org.jboss.netty.handler.codec.compression.ZlibWrapper;
25  
26  /**
27   * Creates a newly configured {@link ChannelPipeline} for a server-side channel.
28   */
29  public class FactorialServerPipelineFactory implements
30          ChannelPipelineFactory {
31  
32      public ChannelPipeline getPipeline() throws Exception {
33          ChannelPipeline pipeline = pipeline();
34  
35          // Enable stream compression (you can remove these two if unnecessary)
36          pipeline.addLast("deflater", new ZlibEncoder(ZlibWrapper.GZIP));
37          pipeline.addLast("inflater", new ZlibDecoder(ZlibWrapper.GZIP));
38  
39          // Add the number codec first,
40          pipeline.addLast("decoder", new BigIntegerDecoder());
41          pipeline.addLast("encoder", new NumberEncoder());
42  
43          // and then business logic.
44          // Please note we create a handler for every new channel
45          // because it has stateful properties.
46          pipeline.addLast("handler", new FactorialServerHandler());
47  
48          return pipeline;
49      }
50  }
