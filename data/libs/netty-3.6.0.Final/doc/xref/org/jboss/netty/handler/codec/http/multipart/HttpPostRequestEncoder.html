117     /**
118      *
119      * @param factory the factory used to create InterfaceHttpData
120      * @param request the request to encode
121      * @param multipart True if the FORM is a ENCTYPE="multipart/form-data"
122      * @param charset the charset to use as default
123      * @throws NullPointerException for request or charset or factory
124      * @throws ErrorDataEncoderException if the request is not a POST
125      */
126     public HttpPostRequestEncoder(HttpDataFactory factory, HttpRequest request,
127             boolean multipart, Charset charset) throws ErrorDataEncoderException {
128         if (factory == null) {
129             throw new NullPointerException("factory");
130         }
131         if (request == null) {
132             throw new NullPointerException("request");
133         }
134         if (charset == null) {
135             throw new NullPointerException("charset");
136         }
137         if (request.getMethod() != HttpMethod.POST) {
138             throw new ErrorDataEncoderException("Cannot create a Encoder if not a POST");
139         }
140         this.request = request;
141         this.charset = charset;
142         this.factory = factory;
143         // Fill default values
144         bodyListDatas = new ArrayList<InterfaceHttpData>();
145         // default mode
146         isLastChunk = false;
147         isLastChunkSent = false;
148         isMultipart = multipart;
149         multipartHttpDatas = new ArrayList<InterfaceHttpData>();
150         if (isMultipart) {
151             initDataMultipart();
152         }
153     }

227      * @throws NullPointerException for datas
228      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
229      */
230     public void setBodyHttpDatas(List<InterfaceHttpData> datas)
231             throws ErrorDataEncoderException {
232         if (datas == null) {
233             throw new NullPointerException("datas");
234         }
235         globalBodySize = 0;
236         bodyListDatas.clear();
237         currentFileUpload = null;
238         duringMixedMode = false;
239         multipartHttpDatas.clear();
240         for (InterfaceHttpData data: datas) {

265     /**
266      * Add a file as a FileUpload
267      * @param name the name of the parameter
268      * @param file the file to be uploaded (if not Multipart mode, only the filename will be included)
269      * @param contentType the associated contentType for the File
270      * @param isText True if this file should be transmitted in Text format (else binary)
271      * @throws NullPointerException for name and file
272      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
273      */
274     public void addBodyFileUpload(String name, File file, String contentType, boolean isText)
275     throws ErrorDataEncoderException {
276         if (name == null) {
277             throw new NullPointerException("name");
278         }
279         if (file == null) {
280             throw new NullPointerException("file");
281         }
282         String scontentType = contentType;
283         String contentTransferEncoding = null;
284         if (contentType == null) {
285             if (isText) {
286                 scontentType = HttpPostBodyUtil.DEFAULT_TEXT_CONTENT_TYPE;
287             } else {
288                 scontentType = HttpPostBodyUtil.DEFAULT_BINARY_CONTENT_TYPE;
289             }
290         }
291         if (!isText) {
292             contentTransferEncoding = HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value();
293         }
294         FileUpload fileUpload = factory.createFileUpload(request, name, file.getName(),
295                 scontentType, contentTransferEncoding, null, file.length());
296         try {
297             fileUpload.setContent(file);
298         } catch (IOException e) {
299             throw new ErrorDataEncoderException(e);
300         }
301         addBodyHttpData(fileUpload);
302     }

304     /**
305      * Add a series of Files associated with one File parameter (implied Mixed mode in Multipart)
306      * @param name the name of the parameter
307      * @param file the array of files
308      * @param contentType the array of content Types associated with each file
309      * @param isText the array of isText attribute (False meaning binary mode) for each file
310      * @throws NullPointerException also throws if array have different sizes
311      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
312      */
313     public void addBodyFileUploads(String name, File[] file, String[] contentType, boolean[] isText)
314     throws ErrorDataEncoderException {
315         if (file.length != contentType.length && file.length != isText.length) {
316             throw new NullPointerException("Different array length");
317         }
318         for (int i = 0; i < file.length; i++) {
319             addBodyFileUpload(name, file[i], contentType[i], isText[i]);
320         }
321     }

323     /**
324      * Add the InterfaceHttpData to the Body list
325      * @throws NullPointerException for data
326      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
327      */
328     public void addBodyHttpData(InterfaceHttpData data)
329     throws ErrorDataEncoderException {
330         if (headerFinalized) {
331             throw new ErrorDataEncoderException("Cannot add value once finalized");
332         }
333         if (data == null) {
334             throw new NullPointerException("data");
335         }
336         bodyListDatas.add(data);
337         if (! isMultipart) {
338             if (data instanceof Attribute) {
339                 Attribute attribute = (Attribute) data;
340                 try {
341                     // name=value& with encoded name and attribute
342                     String key = encodeAttribute(attribute.getName(), charset);
343                     String value = encodeAttribute(attribute.getValue(), charset);
344                     Attribute newattribute = factory.createAttribute(request, key, value);
345                     multipartHttpDatas.add(newattribute);
346                     globalBodySize += newattribute.getName().length() + 1 +
347                         newattribute.length() + 1;
348                 } catch (IOException e) {
349                     throw new ErrorDataEncoderException(e);
350                 }
351             } else if (data instanceof FileUpload) {
352                 // since not Multipart, only name=filename => Attribute
353                 FileUpload fileUpload = (FileUpload) data;
354                 // name=filename& with encoded name and filename
355                 String key = encodeAttribute(fileUpload.getName(), charset);
356                 String value = encodeAttribute(fileUpload.getFilename(), charset);
357                 Attribute newattribute = factory.createAttribute(request, key, value);
358                 multipartHttpDatas.add(newattribute);
359                 globalBodySize += newattribute.getName().length() + 1 +
360                     newattribute.length() + 1;
361             }
362             return;
363         }
364         /*
365          * Logic:
366          * if not Attribute:
367          *      add Data to body list
368          *      if (duringMixedMode)
369          *          add endmixedmultipart delimiter
370          *          currentFileUpload = null
371          *          duringMixedMode = false;
372          *      add multipart delimiter, multipart body header and Data to multipart list
373          *      reset currentFileUpload, duringMixedMode
374          * if FileUpload: take care of multiple file for one field => mixed mode
375          *      if (duringMixeMode)
376          *          if (currentFileUpload.name == data.name)
377          *              add mixedmultipart delimiter, mixedmultipart body header and Data to multipart list
378          *          else
379          *              add endmixedmultipart delimiter, multipart body header and Data to multipart list
380          *              currentFileUpload = data
381          *              duringMixedMode = false;
382          *      else
383          *          if (currentFileUpload.name == data.name)
384          *              change multipart body header of previous file into multipart list to
385          *                      mixedmultipart start, mixedmultipart body header
386          *              add mixedmultipart delimiter, mixedmultipart body header and Data to multipart list
387          *              duringMixedMode = true
388          *          else
389          *              add multipart delimiter, multipart body header and Data to multipart list
390          *              currentFileUpload = data
391          *              duringMixedMode = false;
392          * Do not add last delimiter! Could be:
393          * if duringmixedmode: endmixedmultipart + endmultipart
394          * else only endmultipart
395          */
396         if (data instanceof Attribute) {
397             if (duringMixedMode) {
398                 InternalAttribute internal = new InternalAttribute();
399                 internal.addValue("\r\n--" + multipartMixedBoundary + "--");
400                 multipartHttpDatas.add(internal);
401                 multipartMixedBoundary = null;
402                 currentFileUpload = null;
403                 duringMixedMode = false;
404             }
405             InternalAttribute internal = new InternalAttribute();
406             if (!multipartHttpDatas.isEmpty()) {
407                 // previously a data field so CRLF
408                 internal.addValue("\r\n");
409             }
410             internal.addValue("--" + multipartDataBoundary + "\r\n");
411             // content-disposition: form-data; name="field1"
412             Attribute attribute = (Attribute) data;
413             internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
414                     HttpPostBodyUtil.FORM_DATA + "; " +
415                     HttpPostBodyUtil.NAME + "=\"" +
416                     encodeAttribute(attribute.getName(), charset) + "\"\r\n");
417             Charset localcharset = attribute.getCharset();
418             if (localcharset != null) {
419                 // Content-Type: charset=charset
420                 internal.addValue(HttpHeaders.Names.CONTENT_TYPE + ": " +
421                         HttpHeaders.Values.CHARSET + '=' + localcharset + "\r\n");
422             }
423             // CRLF between body header and data
424             internal.addValue("\r\n");
425             multipartHttpDatas.add(internal);
426             multipartHttpDatas.add(data);
427             globalBodySize += attribute.length() + internal.size();
428         } else if (data instanceof FileUpload) {
429             FileUpload fileUpload = (FileUpload) data;
430             InternalAttribute internal = new InternalAttribute();
431             if (!multipartHttpDatas.isEmpty()) {
432                 // previously a data field so CRLF
433                 internal.addValue("\r\n");
434             }
435             boolean localMixed;
436             if (duringMixedMode) {
437                 if (currentFileUpload != null &&
438                         currentFileUpload.getName().equals(fileUpload.getName())) {
439                     // continue a mixed mode
440 
441                     localMixed = true;
442                 } else {
443                     // end a mixed mode
444 
445                     // add endmixedmultipart delimiter, multipart body header and
446                     // Data to multipart list
447                     internal.addValue("--" + multipartMixedBoundary + "--");
448                     multipartHttpDatas.add(internal);
449                     multipartMixedBoundary = null;
450                     // start a new one (could be replaced if mixed start again from here
451                     internal = new InternalAttribute();
452                     internal.addValue("\r\n");
453                     localMixed = false;
454                     // new currentFileUpload and no more in Mixed mode
455                     currentFileUpload = fileUpload;
456                     duringMixedMode = false;
457                 }
458             } else {
459                 if (currentFileUpload != null &&
460                         currentFileUpload.getName().equals(fileUpload.getName())) {
461                     // create a new mixed mode (from previous file)
462 
463                     // change multipart body header of previous file into multipart list to
464                     // mixedmultipart start, mixedmultipart body header
465 
466                     // change Internal (size()-2 position in multipartHttpDatas)
467                     // from (line starting with *)
468                     // --AaB03x
469                     // * Content-Disposition: form-data; name="files"; filename="file1.txt"
470                     // Content-Type: text/plain
471                     // to (lines starting with *)
472                     // --AaB03x
473                     // * Content-Disposition: form-data; name="files"
474                     // * Content-Type: multipart/mixed; boundary=BbC04y
475                     // *
476                     // * --BbC04y
477                     // * Content-Disposition: file; filename="file1.txt"
478                     // Content-Type: text/plain
479                     initMixedMultipart();
480                     InternalAttribute pastAttribute =
481                         (InternalAttribute) multipartHttpDatas.get(multipartHttpDatas.size() - 2);
482                     // remove past size
483                     globalBodySize -= pastAttribute.size();
484                     String replacement = HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
485                         HttpPostBodyUtil.FORM_DATA + "; " + HttpPostBodyUtil.NAME + "=\"" +
486                         encodeAttribute(fileUpload.getName(), charset) + "\"\r\n";
487                     replacement += HttpHeaders.Names.CONTENT_TYPE + ": " +
488                         HttpPostBodyUtil.MULTIPART_MIXED + "; " + HttpHeaders.Values.BOUNDARY +
489                         '=' + multipartMixedBoundary + "\r\n\r\n";
490                     replacement += "--" + multipartMixedBoundary + "\r\n";
491                     replacement += HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
492                         HttpPostBodyUtil.FILE + "; " + HttpPostBodyUtil.FILENAME + "=\"" +
493                         encodeAttribute(fileUpload.getFilename(), charset) +
494                         "\"\r\n";
495                     pastAttribute.setValue(replacement, 1);
496                     // update past size
497                     globalBodySize += pastAttribute.size();
498 
499                     // now continue
500                     // add mixedmultipart delimiter, mixedmultipart body header and
501                     // Data to multipart list
502                     localMixed = true;
503                     duringMixedMode = true;
504                 } else {
505                     // a simple new multipart
506                     //add multipart delimiter, multipart body header and Data to multipart list
507                     localMixed = false;
508                     currentFileUpload = fileUpload;
509                     duringMixedMode = false;
510                 }
511             }
512 
513             if (localMixed) {
514                 // add mixedmultipart delimiter, mixedmultipart body header and
515                 // Data to multipart list
516                 internal.addValue("--" + multipartMixedBoundary + "\r\n");
517                 // Content-Disposition: file; filename="file1.txt"
518                 internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
519                         HttpPostBodyUtil.FILE + "; " + HttpPostBodyUtil.FILENAME + "=\"" +
520                         encodeAttribute(fileUpload.getFilename(), charset) +
521                         "\"\r\n");
522             } else {
523                 internal.addValue("--" + multipartDataBoundary + "\r\n");
524                 // Content-Disposition: form-data; name="files"; filename="file1.txt"
525                 internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " +
526                         HttpPostBodyUtil.FORM_DATA + "; " + HttpPostBodyUtil.NAME + "=\"" +
527                         encodeAttribute(fileUpload.getName(), charset) + "\"; " +
528                         HttpPostBodyUtil.FILENAME + "=\"" +
529                         encodeAttribute(fileUpload.getFilename(), charset) +
530                         "\"\r\n");
531             }
532             // Content-Type: image/gif
533             // Content-Type: text/plain; charset=ISO-8859-1
534             // Content-Transfer-Encoding: binary
535             internal.addValue(HttpHeaders.Names.CONTENT_TYPE + ": " +
536                     fileUpload.getContentType());
537             String contentTransferEncoding = fileUpload.getContentTransferEncoding();
538             if (contentTransferEncoding != null &&
539                     contentTransferEncoding.equals(
540                             HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value())) {
541                 internal.addValue("\r\n" + HttpHeaders.Names.CONTENT_TRANSFER_ENCODING +
542                         ": " + HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value() +
543                         "\r\n\r\n");
544             } else if (fileUpload.getCharset() != null) {
545                 internal.addValue("; " + HttpHeaders.Values.CHARSET + '=' +
546                         fileUpload.getCharset() + "\r\n\r\n");
547             } else {
548                 internal.addValue("\r\n\r\n");
549             }
550             multipartHttpDatas.add(internal);
551             multipartHttpDatas.add(data);
552             globalBodySize += fileUpload.length() + internal.size();
553         }
554     }

561     /**
562      * Finalize the request by preparing the Header in the request and
563      * returns the request ready to be sent.<br>
564      * Once finalized, no data must be added.<br>
565      * If the request does not need chunk (isChunked() == false),
566      * this request is the only object to send to
567      * the remote server.
568      *
569      * @return the request object (chunked or not according to size of body)
570      * @throws ErrorDataEncoderException if the encoding is in error or if the finalize were already done
571      */
572     public HttpRequest finalizeRequest() throws ErrorDataEncoderException {
573         // Finalize the multipartHttpDatas
574         if (! headerFinalized) {
575             if (isMultipart) {
576                 InternalAttribute internal = new InternalAttribute();
577                 if (duringMixedMode) {
578                     internal.addValue("\r\n--" + multipartMixedBoundary + "--");
579                 }
580                 internal.addValue("\r\n--" + multipartDataBoundary + "--\r\n");
581                 multipartHttpDatas.add(internal);
582                 multipartMixedBoundary = null;
583                 currentFileUpload = null;
584                 duringMixedMode = false;
585                 globalBodySize += internal.size();
586             }
587             headerFinalized = true;
588         } else {
589             throw new ErrorDataEncoderException("Header already encoded");
590         }
591         List<String> contentTypes = request.getHeaders(HttpHeaders.Names.CONTENT_TYPE);
592         List<String> transferEncoding =
593             request.getHeaders(HttpHeaders.Names.TRANSFER_ENCODING);
594         if (contentTypes != null) {
595             request.removeHeader(HttpHeaders.Names.CONTENT_TYPE);
596             for (String contentType: contentTypes) {
597                 // "multipart/form-data; boundary=--89421926422648"
598                 if (contentType.toLowerCase().startsWith(
599                         HttpHeaders.Values.MULTIPART_FORM_DATA)) {
600                     // ignore
601                 } else if (contentType.toLowerCase().startsWith(
602                         HttpHeaders.Values.APPLICATION_X_WWW_FORM_URLENCODED)) {
603                     // ignore
604                 } else {
605                     request.addHeader(HttpHeaders.Names.CONTENT_TYPE, contentType);
606                 }
607             }
608         }
609         if (isMultipart) {
610             String value = HttpHeaders.Values.MULTIPART_FORM_DATA + "; " +
611                 HttpHeaders.Values.BOUNDARY + '=' + multipartDataBoundary;
612             request.addHeader(HttpHeaders.Names.CONTENT_TYPE, value);
613         } else {
614             // Not multipart
615             request.addHeader(HttpHeaders.Names.CONTENT_TYPE,
616                     HttpHeaders.Values.APPLICATION_X_WWW_FORM_URLENCODED);
617         }
618         // Now consider size for chunk or not
619         long realSize = globalBodySize;
620         if (isMultipart) {
621             iterator = multipartHttpDatas.listIterator();
622         } else {
623             realSize -= 1; // last '&' removed
624             iterator = multipartHttpDatas.listIterator();
625         }
626         request.setHeader(HttpHeaders.Names.CONTENT_LENGTH, String
627                 .valueOf(realSize));
628         if (realSize > HttpPostBodyUtil.chunkSize || isMultipart) {
629             isChunked = true;
630             if (transferEncoding != null) {
631                 request.removeHeader(HttpHeaders.Names.TRANSFER_ENCODING);
632                 for (String v: transferEncoding) {
633                     if (v.equalsIgnoreCase(HttpHeaders.Values.CHUNKED)) {
634                         // ignore
635                     } else {
636                         request.addHeader(HttpHeaders.Names.TRANSFER_ENCODING, v);
637                     }
638                 }
639             }
640             request.addHeader(HttpHeaders.Names.TRANSFER_ENCODING,
641                     HttpHeaders.Values.CHUNKED);
642             request.setContent(ChannelBuffers.EMPTY_BUFFER);
643         } else {
644             // get the only one body and set it to the request
645             HttpChunk chunk = nextChunk();
646             request.setContent(chunk.getContent());
647         }
648         return request;
649     }

658     /**
659      * Encode one attribute
660      * @return the encoded attribute
661      * @throws ErrorDataEncoderException if the encoding is in error
662      */
663     private static String encodeAttribute(String s, Charset charset)
664             throws ErrorDataEncoderException {
665         if (s == null) {
666             return "";
667         }
668         try {
669             return URLEncoder.encode(s, charset.name());
670         } catch (UnsupportedEncodingException e) {
671             throw new ErrorDataEncoderException(charset.name(), e);
672         }
673     }

688     /**
689      *
690      * @return the next ChannelBuffer to send as a HttpChunk and modifying currentBuffer
691      * accordingly
692      */
693     private ChannelBuffer fillChannelBuffer() {
694         int length = currentBuffer.readableBytes();
695         if (length > HttpPostBodyUtil.chunkSize) {
696             ChannelBuffer slice =
697                 currentBuffer.slice(currentBuffer.readerIndex(), HttpPostBodyUtil.chunkSize);
698             currentBuffer.skipBytes(HttpPostBodyUtil.chunkSize);
699             return slice;
700         } else {
701             // to continue
702             ChannelBuffer slice = currentBuffer;
703             currentBuffer = null;
704             return slice;
705         }
706     }

708     /**
709      * From the current context (currentBuffer and currentData), returns the next HttpChunk
710      * (if possible) trying to get sizeleft bytes more into the currentBuffer.
711      * This is the Multipart version.
712      *
713      * @param sizeleft the number of bytes to try to get from currentData
714      * @return the next HttpChunk or null if not enough bytes were found
715      * @throws ErrorDataEncoderException if the encoding is in error
716      */
717     private HttpChunk encodeNextChunkMultipart(int sizeleft) throws ErrorDataEncoderException {
718         if (currentData == null) {
719             return null;
720         }
721         ChannelBuffer buffer;
722         if (currentData instanceof InternalAttribute) {
723             String internal = currentData.toString();
724             byte[] bytes;
725             try {
726                 bytes = internal.getBytes("ASCII");
727             } catch (UnsupportedEncodingException e) {
728                 throw new ErrorDataEncoderException(e);
729             }
730             buffer = ChannelBuffers.wrappedBuffer(bytes);
731             currentData = null;
732         } else {
733             if (currentData instanceof Attribute) {
734                 try {
735                     buffer = ((Attribute) currentData).getChunk(sizeleft);
736                 } catch (IOException e) {
737                     throw new ErrorDataEncoderException(e);
738                 }
739             } else {
740                 try {
741                     buffer = ((HttpData) currentData).getChunk(sizeleft);
742                 } catch (IOException e) {
743                     throw new ErrorDataEncoderException(e);
744                 }
745             }
746             if (buffer.capacity() == 0) {
747                 // end for current InterfaceHttpData, need more data
748                 currentData = null;
749                 return null;
750             }
751         }
752         if (currentBuffer == null) {
753             currentBuffer = buffer;
754         } else {
755             currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
756                 buffer);
757         }
758         if (currentBuffer.readableBytes() < HttpPostBodyUtil.chunkSize) {
759             currentData = null;
760             return null;
761         }
762         buffer = fillChannelBuffer();
763         return new DefaultHttpChunk(buffer);
764     }

766     /**
767      * From the current context (currentBuffer and currentData), returns the next HttpChunk
768      * (if possible) trying to get sizeleft bytes more into the currentBuffer.
769      * This is the UrlEncoded version.
770      *
771      * @param sizeleft the number of bytes to try to get from currentData
772      * @return the next HttpChunk or null if not enough bytes were found
773      * @throws ErrorDataEncoderException if the encoding is in error
774      */
775     private HttpChunk encodeNextChunkUrlEncoded(int sizeleft) throws ErrorDataEncoderException {
776         if (currentData == null) {
777             return null;
778         }
779         int size = sizeleft;
780         ChannelBuffer buffer;
781         if (isKey) {
782             // get name
783             String key = currentData.getName();
784             buffer = ChannelBuffers.wrappedBuffer(key.getBytes());
785             isKey = false;
786             if (currentBuffer == null) {
787                 currentBuffer = ChannelBuffers.wrappedBuffer(
788                         buffer, ChannelBuffers.wrappedBuffer("=".getBytes()));
789                 //continue
790                 size -= buffer.readableBytes() + 1;
791             } else {
792                 currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
793                     buffer, ChannelBuffers.wrappedBuffer("=".getBytes()));
794                 //continue
795                 size -= buffer.readableBytes() + 1;
796             }
797             if (currentBuffer.readableBytes() >= HttpPostBodyUtil.chunkSize) {
798                 buffer = fillChannelBuffer();
799                 return new DefaultHttpChunk(buffer);
800             }
801         }
802         try {
803             buffer = ((HttpData) currentData).getChunk(size);
804         } catch (IOException e) {
805             throw new ErrorDataEncoderException(e);
806         }
807         ChannelBuffer delimiter = null;
808         if (buffer.readableBytes() < size) {
809             // delimiter
810             isKey = true;
811             delimiter = iterator.hasNext() ?
812                     ChannelBuffers.wrappedBuffer("&".getBytes()) :
813                         null;
814         }
815         if (buffer.capacity() == 0) {
816             // end for current InterfaceHttpData, need potentially more data
817             currentData = null;
818             if (currentBuffer == null) {
819                 currentBuffer = delimiter;
820             } else {
821                 if (delimiter != null) {
822                     currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
823                         delimiter);
824                 }
825             }
826             if (currentBuffer.readableBytes() >= HttpPostBodyUtil.chunkSize) {
827                 buffer = fillChannelBuffer();
828                 return new DefaultHttpChunk(buffer);
829             }
830             return null;
831         }
832         if (currentBuffer == null) {
833             if (delimiter != null) {
834                 currentBuffer = ChannelBuffers.wrappedBuffer(buffer,
835                     delimiter);
836             } else {
837                 currentBuffer = buffer;
838             }
839         } else {
840             if (delimiter != null) {
841                 currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
842                     buffer, delimiter);
843             } else {
844                 currentBuffer = ChannelBuffers.wrappedBuffer(currentBuffer,
845                         buffer);
846             }
847         }
848         if (currentBuffer.readableBytes() < HttpPostBodyUtil.chunkSize) {
849             // end for current InterfaceHttpData, need more data
850             currentData = null;
851             isKey = true;
852             return null;
853         }
854         buffer = fillChannelBuffer();
855         // size = 0
856         return new DefaultHttpChunk(buffer);
857     }

863     /**
864      * Returns the next available HttpChunk. The caller is responsible to test if this chunk is the
865      * last one (isLast()), in order to stop calling this method.
866      *
867      * @return the next available HttpChunk
868      * @throws ErrorDataEncoderException if the encoding is in error
869      */
870     public HttpChunk nextChunk() throws ErrorDataEncoderException {
871         if (isLastChunk) {
872             isLastChunkSent = true;
873             return new DefaultHttpChunk(ChannelBuffers.EMPTY_BUFFER);
874         }
875         ChannelBuffer buffer;
876         int size = HttpPostBodyUtil.chunkSize;
877         // first test if previous buffer is not empty
878         if (currentBuffer != null) {
879             size -= currentBuffer.readableBytes();
880         }
881         if (size <= 0) {
882             //NextChunk from buffer
883             buffer = fillChannelBuffer();
884             return new DefaultHttpChunk(buffer);
885         }
886         // size > 0
887         if (currentData != null) {
888             // continue to read data
889             if (isMultipart) {
890                 HttpChunk chunk = encodeNextChunkMultipart(size);
891                 if (chunk != null) {
892                     return chunk;
893                 }
894             } else {
895                 HttpChunk chunk = encodeNextChunkUrlEncoded(size);
896                 if (chunk != null) {
897                     //NextChunk Url from currentData
898                     return chunk;
899                 }
900             }
901             size = HttpPostBodyUtil.chunkSize - currentBuffer.readableBytes();
902         }
903         if (! iterator.hasNext()) {
904             isLastChunk = true;
905             //NextChunk as last non empty from buffer
906             buffer = currentBuffer;
907             currentBuffer = null;
908             return new DefaultHttpChunk(buffer);
909         }
910         while (size > 0 && iterator.hasNext()) {
911             currentData = iterator.next();
912             HttpChunk chunk;
913             if (isMultipart) {
914                 chunk = encodeNextChunkMultipart(size);
915             } else {
916                 chunk = encodeNextChunkUrlEncoded(size);
917             }
918             if (chunk == null) {
919                 // not enough
920                 size = HttpPostBodyUtil.chunkSize - currentBuffer.readableBytes();
921                 continue;
922             }
923             //NextChunk from data
924             return chunk;
925         }
926         // end since no more data
927         isLastChunk = true;
928         if (currentBuffer == null) {
929             isLastChunkSent = true;
930             //LastChunk with no more data
931             return new DefaultHttpChunk(ChannelBuffers.EMPTY_BUFFER);
932         }
933         //Previous LastChunk with no more data
934         buffer = currentBuffer;
935         currentBuffer = null;
936         return new DefaultHttpChunk(buffer);
937     }
