86      static int crc32(int crc32, byte[] buf, int index, int len) {
87          final int endIndex = index + len;
88          crc32 ^= 0xffffffff;
89          for (int i = index; i < endIndex; i ++) {
90              crc32 = crc32 >>> 8 ^ TABLE[(crc32 ^ buf[i]) & 0xff];
91          }
92          crc32 ^= 0xffffffff;
93          return crc32;
94      }
